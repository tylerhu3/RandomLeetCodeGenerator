CREATE TABLE inputMeals(
id INT AUTO_INCREMENT,
   Meals VARCHAR(100),
   Actual_Meals VARCHAR(100),
   PRIMARY KEY(id)
);



##get column names from table inputOrders that have substring wkly_spcl_1 

select COLUMN_NAME from information_schema.columns 
where table_name='inputOrders' and column_name like '%wkly_spcl_1%'
